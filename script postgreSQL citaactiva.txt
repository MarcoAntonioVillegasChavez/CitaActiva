	plannedHours INT,
	operatorId VARCHAR(100),
	teamId VARCHAR(100),
	PRIMARY KEY (id)
);

	id VARCHAR(10) NOT NULL,

	contactMail VARCHAR(100) NOT NULL,
	contactPhone VARCHAR(100) NOT NULL,
	brandId VARCHAR(100) NOT NULL,
